17 // NOTE: inputs are transposed versions of algebra X and Z. That is,
18 // X and Z are m x n matrices, where n = length(acf).
19 template <typename T1, typename T2>
20 inline void DurbinLevinsonXZ(const Eigen::MatrixBase<T1>& X,
21  const Eigen::MatrixBase<T2>& Z,
22  const Ref <const VectorXd>& acf,
23  Ref <VectorXd> phi, Ref <VectorXd> phi2, Ref <VectorXd> res,
24  bool ZtoX) {
25 #define _X const_cast<Eigen::MatrixBase<T1>& >(X)
26 #define _Z const_cast<Eigen::MatrixBase<T2>& >(Z)
27  int n; //,m;
28  n = acf.size();
29  //m = X.rows();
30  double nu, sqrtNu, rp;
31  nu = acf(0);
32  int ii; //, jj, kk;
33  for(ii = 0; ii < n; ii++) {
34  // variance
35  if(ii > 0) nu *= (1-phi(ii-1)*phi(ii-1));
36  sqrtNu = sqrt(nu);
37  // mean and actual conversion
38  if(ii == 0) {
39  res.setZero();
40  }
41  else {
42  res = _X.leftCols(ii) * phi.head(ii).reverse();
43  }
44  if(ZtoX) {
45  _X.col(ii) = res + sqrtNu * _Z.col(ii);
46  }
47  else {
48  _Z.col(ii) = (_X.col(ii) - res)/sqrtNu;
49  }
50  // new coefficients
51  if(ii < n-1) {
52  // coefficients
53  phi2.head(ii) = phi.head(ii).reverse();
54  rp = phi2.head(ii).dot(acf.segment(1,ii));
55  //rp = phi.head(ii).dot(acf.segment(1,ii).reverse());
56  //phi2.head(ii) = phi.head(ii).reverse();
57  phi(ii) = (acf(ii+1) - rp)/nu;
58  phi.head(ii) -= phi(ii) * phi2.head(ii);
59  }
60  }

62 #undef _Z
63  return;
64 }
