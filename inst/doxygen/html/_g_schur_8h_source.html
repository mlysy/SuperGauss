19 inline vector<int> int2Bin(int x, int base = 1) {
20  vector<int> s;
21  int x1 = x / base;
22  int dif = x - x1 * base;
23  int M = base;
24  do{
25  if (x1 & 1){
26  s.push_back(M);
27  }
28  M <<= 1;
29  } while ((x1 >>= 1)>0);
30  reverse(s.begin(), s.end());
31  if (dif){
32  s.push_back(dif);
33  }
34  return s;
35 }

225 inline void GSchurN::ProgStep(double* alpha0, double* beta0, int size) {
226  // alpha_FFT->in, beta_FFT->in, xi_m_FFTPS->in, eta_m_FFT->in, xi_IFFT->out, eta_IFFT->out, gamma are n
227  std::fill(gs[0]->xi_m_FFT->in, gs[0]->xi_m_FFT->in + 2*size, 0);
228  std::fill(gs[0]->eta_m_FFT->in, gs[0]->eta_m_FFT->in + 2*size, 0);
229  double *tmpPtr, *xi1, *xi2, *eta1, *eta2, alpha_0, alpha_1;
230  eta1 = gs[0]->eta_m_FFT->in;
231  eta2 = gs[0]->eta_m_FFT->in + size;
232  xi1 = gs[0]->xi_m_FFT->in;
233  xi2 = gs[0]->xi_m_FFT->in + size;
234  eta1[0] = 1.0;
235  xi1[0] = alpha0[0]/beta0[0];
236  gs[0]->gamma[0] = xi1[0];
237  gs[0]->beta_FFT->in[0] = beta0[0] * (1 - xi1[0] * xi1[0]);
238 
239  for (int kk = 1; kk < size; ++kk){
240  alpha_1 = alpha0[kk];
241  gs[0]->beta_FFT->in[kk] = beta0[kk];
242  for (int jj = 1; jj <= kk; ++jj){
243  alpha_0 = alpha_1 - gs[0]->gamma[jj - 1] * gs[0]->beta_FFT->in[kk - jj + 1];
244  gs[0]->beta_FFT->in[kk - jj + 1] -= gs[0]->gamma[jj - 1] * alpha_1;
245  alpha_1 = alpha_0;
246  }
247  gs[0]->gamma[kk] = alpha_0 / gs[0]->beta_FFT->in[0];
248  gs[0]->beta_FFT->in[0] *= 1 - gs[0]->gamma[kk] * gs[0]->gamma[kk];
249 
250  eta2[0] = 1.0;
251  xi2[0] = alpha0[0]/beta0[0];
252  for (int jj = 1; jj <= kk; ++jj){
253  xi2[jj] = xi1[jj] + gs[0]->gamma[kk] * eta1[kk - jj];
254  eta2[jj] = eta1[jj] + gs[0]->gamma[kk] *xi1[kk - jj];
255  }
256  tmpPtr = xi1;
257  xi1 = xi2;
258  xi2 = tmpPtr;
259  tmpPtr = eta1;
260  eta1 = eta2;
261  eta2 = tmpPtr;
262  }
263  std::copy(xi1, xi1 + size, gs[0]->xi_IFFT->out);
264  std::copy(eta1, eta1 + size, gs[0]->eta_IFFT->out);
265  return;
266 }
