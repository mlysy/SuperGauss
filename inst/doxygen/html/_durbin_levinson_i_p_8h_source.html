17 // NOTE: X and Y are passed as d x n and k x n matrices
18 template <typename T1>
19 inline void DurbinLevisonEigen(const Eigen::MatrixBase<T1>& M, double &ldV,
20  const Ref <const MatrixXd>& X, const Ref <const MatrixXd>& Y,
21  const Ref <const VectorXd>& acf,
22  Ref <VectorXd> phi, Ref <VectorXd> phi2,
23  Ref <VectorXd> rx, Ref <VectorXd> ry,
24  int calcMode) {
25 #define _M const_cast<Eigen::MatrixBase<T1>& >(M)
26  int n = acf.size();
27  double nu, rp;
28  // initialize
29  nu = acf(0);
30  ldV = 0.0;
31  _M.setZero();
32  for(int ii = 0; ii < n; ii++) {
33  // variance
34  if(ii > 0) nu *= (1.0-phi(ii-1)*phi(ii-1));
35  ldV += log(nu);
36  // residuals
37  rx = X.leftCols(ii) * phi.head(ii).reverse();
38  rx = X.col(ii) - rx;
39  if(calcMode != 1) {
40  ry = Y.leftCols(ii) * phi.head(ii).reverse();
41  ry = Y.col(ii) - ry;
42  }
43  // inner products
44  if(calcMode == 0) {
45  // matrix inner product (default)
46  rx /= nu;
47  _M.noalias() += rx * ry.transpose();
48  }
49  else if(calcMode == 1) {
50  // square norm (i.e. lower triangular only)
51  rx /= sqrt(nu);
52  _M.template selfadjointView<Eigen::Lower>().rankUpdate(rx);
53  } else {
54  // elementwise inner product (i.e. diag(<X, Y>) )
55  _M.array() += (rx.array() * ry.array())/nu;
56  }
57  // new coefficients
58  if(ii < n-1) {
59  // coefficients
60  phi2.head(ii) = phi.head(ii).reverse();
61  rp = phi2.head(ii).dot(acf.segment(1,ii));
62  //rp = phi.head(ii).dot(acf.segment(1,ii).reverse());
63  //phi2.head(ii) = phi.head(ii).reverse();
64  phi(ii) = (acf(ii+1) - rp)/nu;
65  phi.head(ii) -= phi(ii) * phi2.head(ii);
66  }
67  }
68  if(calcMode == 1) {
69  // upper triangular part of M
70  _M.template triangularView<Eigen::Upper>() = M.transpose();
71  }
72  //std::cout << M << std::endl;
73 #undef _M
74  return;
75 }

77 // Base version
78 // NOTE: X and Y are passed as n x d and n x k matrices,
79 // but in columnwise pure double format
80 inline void DurbinLevinsonBase(double *M, double &ldV, double *X,
81  double *Y, double *acf,
82  double *phi, double *phi2, double *rx,
83  double *ry, int n, int d, int k, int calcMode) {
84  int xI, yJ, ii, jj;
85  double nu, rp;
86  nu = acf[0];
87  ldV = 0.0;
88  for(ii = 0; ii < n; ii++) {
89  // variance
90  if(ii > 0) nu *= (1-phi[ii-1]*phi[ii-1]);
91  ldV += log(nu);
92  // residuals
93  for(xI = 0; xI < d; xI++) rx[xI] = 0.0;
94  for(yJ = 0; yJ < k; yJ++) ry[yJ] = 0.0;
95  for(jj = 0; jj < ii; jj++) {
96  for(xI = 0; xI < d; xI++) rx[xI] += phi[jj] * X[xI*n + ii-1-jj];
97  for(yJ = 0; yJ < k; yJ++) ry[yJ] += phi[jj] * Y[yJ*n + ii-1-jj];
98  }
99  for(xI = 0; xI < d; xI++) rx[xI] = X[xI*n + ii] - rx[xI];
100  for(yJ = 0; yJ < k; yJ++) ry[yJ] = Y[yJ*n + ii] - ry[yJ];
101  // inner products
102  if(calcMode == 0) {
103  // matrix inner product (default)
104  for(yJ = 0; yJ < k; yJ++) {
105  for(xI = 0; xI < d; xI++) M[yJ*d + xI] += rx[xI]*ry[yJ]/nu;
106  }
107  }
108  else if(calcMode == 1) {
109  // square norm (i.e. lower triangular only)
110  for(yJ = 0; yJ < d; yJ++) {
111  M[yJ*d + yJ] += rx[yJ]*rx[yJ]/nu;
112  for(xI = 0; xI < yJ; xI++) {
113  M[yJ*d + xI] += rx[xI]*ry[yJ]/nu;
114  }
115  }
116  } else {
117  // elementwise inner product (i.e. diag(<X, Y>) )
118  for(xI = 0; xI < d; xI++) M[xI] += rx[xI]*ry[xI]/nu;
119  }
120  // new coefficients
121  if(ii < n-1) {
122  // coefficients
123  rp = 0.0;
124  for(jj = 0; jj < ii; jj++) {
125  rp += phi[jj] * acf[ii-jj];
126  phi2[jj] = phi[ii-1-jj];
127  }
128  phi[ii] = (acf[ii+1] - rp)/nu;
129  for(jj = 0; jj < ii; jj++) phi[jj] -= phi[ii]*phi2[jj];
130  }
131  }
132  if(calcMode == 1) {
133  for(yJ = 0; yJ < d; yJ++) {
134  for(xI = 0; xI < yJ; xI++) M[xI*d + yJ] = M[yJ*d + xI];
135  }
136  }
137 }
